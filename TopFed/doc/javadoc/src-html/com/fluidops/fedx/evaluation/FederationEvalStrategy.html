123            @Override
124            public CloseableIteration<BindingSet, QueryEvaluationException> evaluate(
125                            TupleExpr expr, BindingSet bindings)
126                            throws QueryEvaluationException {
127                    
128                    if (expr instanceof StatementTupleExpr) {
129                            return ((StatementTupleExpr)expr).evaluate(bindings);
130                    }
131                                    
132                    if (expr instanceof NJoin) {
133                            return evaluateNJoin((NJoin)expr, bindings);
134                    } 
135                    
136                    if (expr instanceof NUnion) {
137                            return evaluateNaryUnion((NUnion)expr, bindings);
138                    }
139                    
140                    if (expr instanceof ExclusiveGroup) {
141                            return ((ExclusiveGroup)expr).evaluate(bindings);
142                    }
143                    
144                    if (expr instanceof SingleSourceQuery)
145                            return evaluateSingleSourceQuery((SingleSourceQuery)expr, bindings);
146                            
147                    if (expr instanceof FedXService) {
148                            return evaluateService((FedXService)expr, bindings);
149                    }
150                    
151                    if (expr instanceof IndependentJoinGroup) {
152                            return evaluateIndependentJoinGroup((IndependentJoinGroup)expr, bindings);      // XXX
153                    }
154                    
155                    if(expr instanceof EmptyResult)
156                            return new EmptyIteration<BindingSet, QueryEvaluationException>();
157                    
158                    return super.evaluate(expr, bindings);
159            }

164            /**
165             * Retrieve the statements matching the provided subject, predicate and object value from the 
166             * federation members.<p>
167             * 
168             * For a bound statement, i.e. a statement with no free variables, the statement itself is 
169             * returned if some member has this statement, an empty iteration otherwise.<p>
170             * 
171             * If the statement has free variables, i.e. one of the provided arguments in <code>null</code>,
172             * the union of results from relevant statement sources is constructed.
173             * 
174             * @param subj
175             * @param pred
176             * @param obj
177             * @param contexts
178             * @return
179             * 
180             * @throws RepositoryException
181             * @throws MalformedQueryException
182             * @throws QueryEvaluationException
183             */
184            public CloseableIteration<Statement, QueryEvaluationException> getStatements(QueryInfo queryInfo, Resource subj, URI pred, Value obj, Resource... contexts) throws RepositoryException, MalformedQueryException, QueryEvaluationException {
185    
186                    if (contexts.length!=0)
187                            log.warn("Context queries are not yet supported by FedX.");
188                    
189                    List<Endpoint> members = FederationManager.getInstance().getFederation().getMembers();
190                    
191                    
192                    // a bound query: if at least one fed member provides results
193                    // return the statement, otherwise empty result
194                    if (subj!=null && pred!=null && obj!=null) {
195                            if (CacheUtils.checkCacheUpdateCache(cache, members, subj, pred, obj)) {
196                                    return new SingletonIteration<Statement, QueryEvaluationException>(new StatementImpl(subj, pred, obj));
197                            }
198                            return new EmptyIteration<Statement, QueryEvaluationException>();
199                    }
200                    
201                    // form the union of results from relevant endpoints
202                    List<StatementSource> sources = CacheUtils.checkCacheForStatementSourcesUpdateCache(cache, members, subj, pred, obj);
203                    
204                    if (sources.size()==0)
205                            return new EmptyIteration<Statement, QueryEvaluationException>();
206                    
207                    if (sources.size()==1) {
208                            Endpoint e = EndpointManager.getEndpointManager().getEndpoint(sources.get(0).getEndpointID());
209                            return e.getTripleSource().getStatements(e.getConn(), subj, pred, obj, contexts);
210                    }
211                    
212                    // TODO why not collect in parallel?
213                    WorkerUnionBase<Statement> union = new SynchronousWorkerUnion<Statement>(queryInfo);                
214                    
215                    for (StatementSource source : sources) {
216                            Endpoint e = EndpointManager.getEndpointManager().getEndpoint(source.getEndpointID());
217                            ParallelGetStatementsTask task = new ParallelGetStatementsTask(union, e.getTripleSource(), e.getConn(), subj, pred, obj, contexts);
218                            union.addTask(task);
219                    }
220                    
221                    // run the union in a separate thread
222                    executor.execute(union);
223                    
224                    // TODO distinct iteration ?
225                    
226                    return union;
227            }

231                    
232                    ParallelServiceExecutor pe = new ParallelServiceExecutor(service, this, bindings);

238            public CloseableIteration<BindingSet, QueryEvaluationException> evaluateSingleSourceQuery(SingleSourceQuery query, BindingSet bindings) throws QueryEvaluationException {
239                    
240                    try
241                    {
242                            Endpoint source = query.getSource();            
243                            return source.getTripleSource().getStatements(query.getQueryString(), source.getConn(), query.getQueryInfo().getQueryType());
244                    } catch (RepositoryException e) {
245                            throw new QueryEvaluationException(e);
246                    } catch (MalformedQueryException e)     {
247                            throw new QueryEvaluationException(e);
248                    }

252            public CloseableIteration<BindingSet, QueryEvaluationException> evaluateNJoin(NJoin join, BindingSet bindings) throws QueryEvaluationException {
253                    
254                    CloseableIteration<BindingSet, QueryEvaluationException> result = evaluate(join.getArg(0), bindings);
255                    
256                    ControlledWorkerScheduler<BindingSet> joinScheduler = FederationManager.getInstance().getJoinScheduler();
257                    
258                    for (int i = 1, n = join.getNumberOfArguments(); i < n; i++) {
259                            result = executeJoin(joinScheduler, result, join.getArg(i), bindings, join.getQueryInfo());
260                    }
261                    return result;
262            }
263            
264            public CloseableIteration<BindingSet, QueryEvaluationException> evaluateNaryUnion(NUnion union, BindingSet bindings) throws QueryEvaluationException {
265                    
266                    ControlledWorkerScheduler<BindingSet> unionScheduler = FederationManager.getInstance().getUnionScheduler();
267                    ControlledWorkerUnion<BindingSet> unionRunnable = new ControlledWorkerUnion<BindingSet>(unionScheduler, union.getQueryInfo());
268                    
269                    for (int i=0; i<union.getNumberOfArguments(); i++) {
270                            unionRunnable.addTask(new ParallelUnionOperatorTask(unionRunnable, this, union.getArg(i), bindings));
271                    }
272                    
273                    executor.execute(unionRunnable);
274                    
275                    return unionRunnable;
276            }

337            public Value evaluate(ValueExpr expr, BindingSet bindings) throws ValueExprEvaluationException, QueryEvaluationException {
338                    
339                    if (expr instanceof FilterExpr)
340                            return evaluate((FilterExpr)expr, bindings);
341                    if (expr instanceof ConjunctiveFilterExpr)
342                            return evaluate((ConjunctiveFilterExpr)expr, bindings);
343                    
344                    return super.evaluate(expr, bindings);
345            }
346            
347            public Value evaluate(FilterExpr node, BindingSet bindings) throws ValueExprEvaluationException, QueryEvaluationException {
348                    
349                    Value v = evaluate(node.getExpression(), bindings);
350                    return BooleanLiteralImpl.valueOf(QueryEvaluationUtil.getEffectiveBooleanValue(v));
351            }
352            
353            
354            public Value evaluate(ConjunctiveFilterExpr node, BindingSet bindings) throws ValueExprEvaluationException, QueryEvaluationException {
355                    
356                    ValueExprEvaluationException error = null;
357                    
358                    for (FilterExpr expr : node.getExpressions()) {
359                            
360                            try {
361                                    Value v = evaluate(expr.getExpression(), bindings);
362                                    if (QueryEvaluationUtil.getEffectiveBooleanValue(v) == false) {
363                                            return BooleanLiteralImpl.FALSE;
364                                    }
365                            } catch (ValueExprEvaluationException e) {
366                                    error = e;
367                            }
368                    }
369                    
370                    if (error!=null)
371                            throw error;
372                    
373                    return BooleanLiteralImpl.TRUE;
374            }

378            protected CloseableIteration<BindingSet, QueryEvaluationException> evaluateAtStatementSources(Object preparedQuery, List<StatementSource> statementSources, QueryInfo queryInfo) throws QueryEvaluationException {
379                    if (preparedQuery instanceof String)
380                            return evaluateAtStatementSources((String)preparedQuery, statementSources, queryInfo);
381                    if (preparedQuery instanceof TupleExpr)
382                            return evaluateAtStatementSources((TupleExpr)preparedQuery, statementSources, queryInfo);
383                    throw new RuntimeException("Unsupported type for prepared query: " + preparedQuery.getClass().getCanonicalName());
384            }
385            
386            protected CloseableIteration<BindingSet, QueryEvaluationException> evaluateAtStatementSources(String preparedQuery, List<StatementSource> statementSources, QueryInfo queryInfo) throws QueryEvaluationException {
387                    
388                    try {
389                            CloseableIteration<BindingSet, QueryEvaluationException> result;
390                            
391                            if (statementSources.size()==1) {                               
392                                    Endpoint ownedEndpoint = EndpointManager.getEndpointManager().getEndpoint(statementSources.get(0).getEndpointID());
393                                    RepositoryConnection conn = ownedEndpoint.getConn();
394                                    com.fluidops.fedx.evaluation.TripleSource t = ownedEndpoint.getTripleSource();
395                                    result = t.getStatements(preparedQuery, conn, EmptyBindingSet.getInstance(), null);
396                            } 
397                             
398                            else {                  
399                                    WorkerUnionBase<BindingSet> union = FederationManager.getInstance().createWorkerUnion(queryInfo);
400                                    
401                                    for (StatementSource source : statementSources) {                                       
402                                            Endpoint ownedEndpoint = EndpointManager.getEndpointManager().getEndpoint(source.getEndpointID());
403                                            RepositoryConnection conn = ownedEndpoint.getConn();
404                                            com.fluidops.fedx.evaluation.TripleSource t = ownedEndpoint.getTripleSource();
405                                            union.addTask(new ParallelPreparedUnionTask(union, preparedQuery, t, conn, EmptyBindingSet.getInstance(), null));
406                                    }
407                                    
408                                    union.run();                            
409                                    result = union;
410                                    
411                                    // TODO we should add some DISTINCT here to have SET semantics
412                            }
413                    
414                            return result;
415                            
416                    } catch (Exception e) {
417                            throw new QueryEvaluationException(e);
418                    }
419            }

422            protected CloseableIteration<BindingSet, QueryEvaluationException> evaluateAtStatementSources(TupleExpr preparedQuery, List<StatementSource> statementSources, QueryInfo queryInfo) throws QueryEvaluationException {
423                    
424                    try {
425                            CloseableIteration<BindingSet, QueryEvaluationException> result;
426                            
427                            if (statementSources.size()==1) {                               
428                                    Endpoint ownedEndpoint = EndpointManager.getEndpointManager().getEndpoint(statementSources.get(0).getEndpointID());
429                                    RepositoryConnection conn = ownedEndpoint.getConn();
430                                    com.fluidops.fedx.evaluation.TripleSource t = ownedEndpoint.getTripleSource();
431                                    result = t.getStatements(preparedQuery, conn, EmptyBindingSet.getInstance(), null);
432                            } 
433                             
434                            else {                  
435                                    WorkerUnionBase<BindingSet> union = FederationManager.getInstance().createWorkerUnion(queryInfo);
436                                    
437                                    for (StatementSource source : statementSources) {                                       
438                                            Endpoint ownedEndpoint = EndpointManager.getEndpointManager().getEndpoint(source.getEndpointID());
439                                            RepositoryConnection conn = ownedEndpoint.getConn();
440                                            com.fluidops.fedx.evaluation.TripleSource t = ownedEndpoint.getTripleSource();
441                                            union.addTask(new ParallelPreparedAlgebraUnionTask(union, preparedQuery, t, conn, EmptyBindingSet.getInstance(), null));                                        
442                                    }
443                                    
444                                    union.run();                            
445                                    result = union;
446                                    
447                                    // TODO we should add some DISTINCT here to have SET semantics
448                            }
449                    
450                            return result;
451                            
452                    } catch (Exception e) {
453                            throw new QueryEvaluationException(e);
454                    }
455            }
