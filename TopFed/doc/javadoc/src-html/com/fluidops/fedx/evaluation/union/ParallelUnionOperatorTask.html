018    package com.fluidops.fedx.evaluation.union;
019    
020    import info.aduna.iteration.CloseableIteration;
021    
022    import org.openrdf.query.BindingSet;
023    import org.openrdf.query.QueryEvaluationException;
024    import org.openrdf.query.algebra.TupleExpr;
025    
026    import com.fluidops.fedx.evaluation.FederationEvalStrategy;
027    import com.fluidops.fedx.evaluation.concurrent.ParallelExecutor;
028    import com.fluidops.fedx.evaluation.concurrent.ParallelTask;
029    
030    /**
031     * A task implementation representing a UNION operator expression to be evaluated.
032     * 
033     * @author Andreas Schwarte
034     */
035    public class ParallelUnionOperatorTask implements ParallelTask<BindingSet> {
036    
037            protected final ParallelExecutor<BindingSet> unionControl;
038            protected final FederationEvalStrategy strategy;
039            protected final TupleExpr expr;
040            protected final BindingSet bindings;
041            
042            public ParallelUnionOperatorTask(ParallelExecutor<BindingSet> unionControl,
043                            FederationEvalStrategy strategy, TupleExpr expr, BindingSet bindings) {
044                    super();
045                    this.unionControl = unionControl;
046                    this.strategy = strategy;
047                    this.expr = expr;
048                    this.bindings = bindings;
049            }
050            
051            @Override
052            public ParallelExecutor<BindingSet> getControl() {
053                    return unionControl;
054            }
055    
056            @Override
057            public CloseableIteration<BindingSet, QueryEvaluationException> performTask()
058                            throws Exception {
059                    return strategy.evaluate(expr, bindings);
060            }
061    }
