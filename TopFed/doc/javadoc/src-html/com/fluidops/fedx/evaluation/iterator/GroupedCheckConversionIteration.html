018    package com.fluidops.fedx.evaluation.iterator;
019    
020    import info.aduna.iteration.CloseableIteration;
021    import info.aduna.iteration.ConvertingIteration;
022    
023    import java.util.Iterator;
024    import java.util.List;
025    
026    import org.openrdf.query.Binding;
027    import org.openrdf.query.BindingSet;
028    import org.openrdf.query.QueryEvaluationException;
029    
030    /**
031     * Retrieves the original bindings for the particular result
032     * 
033     * @author Andreas Schwarte
034     */
035    public class GroupedCheckConversionIteration extends ConvertingIteration<BindingSet, BindingSet, QueryEvaluationException>{
036    
037            protected final List<BindingSet> bindings;
038            
039            public GroupedCheckConversionIteration(CloseableIteration<BindingSet, QueryEvaluationException> iter, List<BindingSet> bindings) {
040                    super(iter);
041                    this.bindings = bindings;
042            }
043    
044            @Override
045            protected BindingSet convert(BindingSet bIn) throws QueryEvaluationException {
046                    int bIndex = -1;
047                    Iterator<Binding> bIter = bIn.iterator();
048                    while (bIter.hasNext()) {
049                            Binding b = bIter.next();
050                            String name = b.getName();
051                            bIndex = Integer.parseInt(name.substring(name.lastIndexOf("_")+1));
052                    }
053                    return bindings.get(bIndex);
054            }
055    }
