059    /**
060     * The FederationManager manages all modules necessary for the runtime behavior. This includes for 
061     * instance the federation layer instance, cache and statistics. It is Singleton and there can only 
062     * be on federation instance at a time.<p>
063     * 
064     * The factory {@link FedXFactory} provides various functions for initialization of FedX and should 
065     * be used as the entry point for any application using FedX.<p>
066     * 
067     * <code>
068     * Config.initialize(fedxConfig);
069     * List&ltEndpoint&gt members = ...                     // e.g. use EndpointFactory methods
070     * FedX fed = FedXFactory.initializeFederation(members);
071     * SailRepository repo = new SailRepository(fed);
072     * ReositoryConnection conn = repo.getConnection();
073     * 
074     * // Do something with the connection, e.g. query evaluation
075     * </code>
076     * 
077     * @author Andreas Schwarte
078     *
079     */
080    public class FederationManager {

097            /**
098             * Initialize the Singleton {@link FederationManager} instance with the provided information. The
099             * {@link FederationManager} remains initialized until {@link #shutDown()} is invoked. 
100             * 
101             * @param members
102             *                              initialize the federation with a list of repository members, null and empty lists are allowed
103             * @param cache
104             *                              the cache instance to be used
105             * @param statistics
106             *                              the statistics instance to be used
107             */
108            public static SailRepository initialize(List<Endpoint> members, Cache cache, Statistics statistics) {
109                    if (instance!=null)
110                            throw new FedXRuntimeException("FederationManager already initialized.");
111                    
112                    log.info("Initializing federation manager ...");
113                    log.info("FedX Version Information: " + Version.getVersionString());            
114    
115                    monitoring = MonitoringFactory.createMonitoring();              
116                    
117                    Executor ex = Executors.newCachedThreadPool();
118                    FedX federation = new FedX(members);
119    
120                    FedXSailRepository repo = new FedXSailRepository(federation);
121                    
122                    instance = new FederationManager(federation, cache, statistics, ex, repo);
123                    instance.updateStrategy();
124                    instance.reset();               
125                    
126                    try     {
127                            repo.initialize();
128                    } catch (RepositoryException e) {
129                            // should never occur
130                            throw new FedXRuntimeException(e);      
131                    }
132                    
133                    EndpointManager.initialize(members);
134                    
135                    // initialize prefix declarations, if any
136                    String prefixFile = Config.getConfig().getPrefixDeclarations();
137                    if (prefixFile!=null) {
138                            QueryManager qm = instance.getQueryManager();
139                            Properties props = new Properties();
140                            try     {
141                                    props.load(new FileInputStream(new File(prefixFile)));
142                            } catch (IOException e) {
143                                    throw new FedXRuntimeException("Error loading prefix properties: " + e.getMessage());
144                            }
145                            
146                            for (String ns : props.stringPropertyNames()) {
147                                    qm.addPrefixDeclaration(ns, props.getProperty(ns));     // register namespace/prefix pair
148                            }
149                    }
150                    
151                    return repo;
152            }

215            /**
216             * Reset the {@link Scheduler} instances, i.e. abort all running threads and create a new
217             * scheduler instance.
218             */
219            public void reset() {
220                    if (log.isDebugEnabled()) {
221                            log.debug("Scheduler for join and union operations initialized with " + Config.getConfig().getJoinWorkerThreads() + " and " + Config.getConfig().getUnionWorkerThreads() + " worker threads, respectively.");
222                    }
223                    
224                    if (joinScheduler!=null)
225                            joinScheduler.abort();
226                    joinScheduler = new ControlledWorkerScheduler<BindingSet>(Config.getConfig().getJoinWorkerThreads(), "Join Scheduler");           
227                    
228                    if (unionScheduler!=null)
229                            unionScheduler.abort();
230                    unionScheduler = new ControlledWorkerScheduler<BindingSet>(Config.getConfig().getUnionWorkerThreads(), "Union Scheduler");                
231                    
232            }

277            /**
278             * Add the specified endpoint to the federation. The endpoint must be initialized and
279             * the federation must not contain a member with the same endpoint location.
280             * 
281             * @param e
282             *                      the initialized endpoint
283             * @param updateStrategy
284             *                      optional parameter, to determine if strategy is to be updated, default=true
285             * 
286             * @throws FedXRuntimeException
287             *                       if the endpoint is not initialized, or if the federation has already a member with the
288             *           same location      
289             */
290            public void addEndpoint(Endpoint e, boolean ...updateStrategy) throws FedXRuntimeException {
291                    log.info("Adding endpoint " + e.getId() + " to federation ...");
292                    
293                    /* check if endpoint is initialized*/
294                    if (!e.isInitialized()) {
295                            try     {
296                                    e.initialize();
297                            } catch (RepositoryException e1){
298                                    throw new FedXRuntimeException("Provided endpoint was not initialized and could not be initialized: " + e1.getMessage(), e1);
299                            }
300                    }
301                    
302                    /* check for duplicate before adding: heuristic => same location */
303                    for (Endpoint member : federation.getMembers())
304                            if (member.getEndpoint().equals(e.getEndpoint()))
305                                    throw new FedXRuntimeException("Adding failed: there exists already an endpoint with location " + e.getEndpoint() + " (eid=" + member.getId() + ")");
306            
307                    federation.addMember(e);
308                    EndpointManager.getEndpointManager().addEndpoint(e);
309                    
310                    if (updateStrategy==null || updateStrategy.length==0 || (updateStrategy.length==1 && updateStrategy[0]==true))
311                            updateStrategy();
312            }
313            
314            /**
315             * Add the specified endpoints to the federation and take care for updating all structures.
316             * 
317             * @param endpoints
318             *                              a list of initialized endpoints to add
319             */
320            public void addAll(List<Endpoint> endpoints) {
321                    log.info("Adding " + endpoints.size() + " endpoints to the federation.");
322                    
323                    for (Endpoint e : endpoints) {
324                            addEndpoint(e, false);
325                    }
326                    
327                    updateStrategy();
328            }
329            
330            /**
331             * Remove the specified endpoint from the federation.
332             * 
333             * @param e
334             *                      the endpoint
335             * @param updateStrategy
336             *                      optional parameter, to determine if strategy is to be updated, default=true
337             */
338            public void removeEndpoint(Endpoint e, boolean ...updateStrategy) throws RepositoryException {
339                    log.info("Removing endpoint " + e.getId() + " from federation ...");
340                    
341                    /* check if e is a federation member */
342                    if (!federation.getMembers().contains(e))
343                            throw new FedXRuntimeException("Endpoint " + e.getId() + " is not a member of the current federation.");
344                    
345                    federation.removeMember(e);
346                    EndpointManager.getEndpointManager().removeEndpoint(e);
347                    e.shutDown();
348                    
349                    if (updateStrategy==null || updateStrategy.length==0 || (updateStrategy.length==1 && updateStrategy[0]==true))
350                            updateStrategy();
351            }
352            
353            /**
354             * Remove all endpoints from the federation, e.g. to load a new preset. Repositories
355             * of the endpoints are shutDown, and the EndpointManager is added accordingly.
356             * 
357             * @throws RepositoryException
358             */
359            public void removeAll() throws RepositoryException {
360                    log.info("Removing all endpoints from federation.");
361                    
362                    for (Endpoint e : new ArrayList<Endpoint>(federation.getMembers())) {
363                            removeEndpoint(e, false);
364                    }
365                    
366                    updateStrategy();
367            }

369            /**
370             * return the number of triples in the federation as string. Retrieving
371             * the size is only supported {@link EndpointType#NativeStore} and
372             * {@link EndpointType#RemoteRepository}.
373             * 
374             * If the federation contains other types of endpoints, the size is
375             * indicated as a lower bound, i.e. the string starts with a 
376             * larger sign.
377             * @return
378             */
379            public String getFederationSize() {
380                    long size = 0;
381                    boolean isLowerBound = false;
382                    for (Endpoint e : getFederation().getMembers())
383                            try     {
384                                    size += e.size();
385                            } catch (RepositoryException e1) {
386                                    isLowerBound = true;
387                            }
388                    return isLowerBound ? ">" + size : Long.toString(size);
389            }

391            /**
392             * Shutdown the federation including the following operations: <p>
393             * 
394             * <ul>
395             *  <li>shut down repositories of all federation members</li>
396             *  <li>persist the cached information</li>
397             *  <li>clear the endpoint manager</li>
398             * </ul>
399             * 
400             * @throws FedXException
401             *                              if an error occurs while shutting down the federation
402             */
403            public void shutDown() throws FedXException {
404                    log.info("Shutting down federation and all underlying repositories ...");
405                    // Abort all running queries
406                    federation.shutDownInternal();
407                    cache.persist();
408                    Config.reset();
409                    EndpointManager.getEndpointManager().shutDown();
410                    instance = null;
411                    monitoring = null;
412            }

414            /**
415             * Create an appropriate worker union for this federation, i.e. a synchronous
416             * worker union for local federations and a multithreaded worker union
417             * for remote & hybrid federations.
418             * 
419             * @return
420             * 
421             * @see ControlledWorkerUnion
422             * @see SynchronousWorkerUnion
423             */
424            public WorkerUnionBase<BindingSet> createWorkerUnion(QueryInfo queryInfo) {
425                    if (type==FederationType.LOCAL)
426                            return new SynchronousWorkerUnion<BindingSet>(queryInfo);
427                    return new ControlledWorkerUnion<BindingSet>(unionScheduler, queryInfo);
428                    
429            }

432            /**
433             * Update the federation evaluation strategy using the classification of endpoints 
434             * as provided by {@link Endpoint#getEndpointClassification()}:<p>
435             * 
436             * <ul>
437             *  <li>local federation: {@link RepositoryFederationEvalStrategy}</li>
438             *  <li>endpoint federation: {@link SparqlFederationEvalStrategy}</li>
439             *  <li>hybrid federation: {@link SparqlFederationEvalStrategy}</li>
440             * </ul>
441             * 
442             */
443            protected void updateStrategy() {
444                    
445                    int localCount=0, remoteCount=0;
446                    for (Endpoint e : federation.getMembers()) {                    
447                            if (e.getEndpointClassification()==EndpointClassification.Remote) 
448                                    remoteCount++;
449                            else 
450                                    localCount++;           
451                    }       
452                    
453                    if (remoteCount==0) {
454                            if (type!=FederationType.LOCAL) {
455                                    type = FederationType.LOCAL;
456                                    strategy = new RepositoryFederationEvalStrategy();              // TODO singleton?
457                            }
458                    } else if (localCount==0) {
459                            if (type!=FederationType.REMOTE) {
460                                    type = FederationType.REMOTE;
461                                    strategy = new SparqlFederationEvalStrategy();
462                            }
463                    } else {
464                            if (type!=FederationType.HYBRID) {
465                                    type = FederationType.HYBRID;
466                                    strategy = new SparqlFederationEvalStrategy();
467                            }                       
468                    }
469                    
470                    log.info("Federation updated. Type: " + type + ", evaluation strategy is " + instance.strategy.getClass().getSimpleName());
471            }
