018    package com.fluidops.fedx.algebra;
019    
020    import org.openrdf.query.algebra.QueryModelVisitor;
021    import org.openrdf.query.algebra.StatementPattern;
022    
023    /**
024     * EmptyStatementPattern represents a statement that cannot produce any results 
025     * for the registered endpoints.
026     * 
027     * @author Andreas Schwarte
028     *
029     */
030    public class EmptyStatementPattern extends StatementPattern implements EmptyResult {
031    
032            public EmptyStatementPattern(StatementPattern node) {
033                    super(node.getSubjectVar(), node.getPredicateVar(), node.getObjectVar(), node.getContextVar());
034            }
035            
036            public <X extends Exception> void visit(QueryModelVisitor<X> visitor) throws X {
037                    visitor.meetOther(this);
038            }
039    }
