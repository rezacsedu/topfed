054    /**
055     * The command line interface for federated query processing with FedX.
056     * 
057     * <code>
058     * Usage:
059     * > FedX [Configuration] [Federation Setup] [Output] [Queries]
060     * > FedX -{help|version}
061     * 
062     * WHERE
063     * [Configuration] (optional)
064     * Optionally specify the configuration to be used
065     *      -c path/to/fedxConfig
066     *      -verbose {0|1|2|3}
067     *      -logtofile
068     *      -p path/to/prefixConfig
069     *      
070     * [Federation Setup] (optional)
071     * Specify one or more federation members
072     *      -s urlToSparqlEndpoint
073     *      -l path/to/NativeStore
074     *      -d path/to/dataconfig.ttl
075     *      
076     * [Output] (optional)
077     * Specify the output options, default stdout. Files are created per query to results/%outputFolder%/q_%id%.{json|xml},
078     * where the outputFolder is the current timestamp, if not specified otherwise.
079     *      -f {STDOUT,JSON,XML}
080     *      -folder outputFolder
081     *      
082     * [Queries]
083     * Specify one or more queries, in file: separation of queries by empty line
084     *      -q sparqlquery
085     *      @q path/to/queryfile
086     *      
087     * Notes:
088     * The federation members can be specified explicitly (-s,-l,-d) or implicitly as 'dataConfig' 
089     * via the fedx configuration  (-f)
090     * 
091     * If no PREFIX declarations are specified in the configurations, the CLI provides
092     * some common PREFIXES, currently rdf, rdfs and foaf. 
093     * </code>
094     * 
095     * 
096     * @author Andreas Schwarte
097     *
098     */
099    public class CLI {

107            protected List<Endpoint> endpoints = new ArrayList<Endpoint>();
108            protected OutputFormat outputFormat = OutputFormat.STDOUT;
109            protected List<String> queries = new ArrayList<String>();

122                            System.exit(1);

128            public void run(String[] args) {
129                    configureRootLogger();
130                    
131                    System.out.println("FedX Cli " + Version.getLongVersion());
132                    
133                    // parse the arguments and construct config
134                    parse(args);
135                    
136                    // activate logging to stdout if verbose is set
137                    configureLogging();
138                                                    
139                    
140                    if (Config.getConfig().getDataConfig()!=null) {
141                            // currently there is no duplicate detection, so the following is a hint for the user
142                            // can cause problems if members are explicitly specified (-s,-l,-d) and via the fedx configuration
143                            if (endpoints.size()>0)
144                                    System.out.println("WARN: Mixture of implicitely and explicitely specified federation members, dataConfig used: " + Config.getConfig().getDataConfig());
145                            try {
146                                    List<Endpoint> additionalEndpoints = EndpointFactory.loadFederationMembers(new File(Config.getConfig().getDataConfig()));
147                                    endpoints.addAll(additionalEndpoints);
148                            } catch (FedXException e) {
149                                    error("Failed to load implicitly specified data sources from fedx configuration. Data config is: " + Config.getConfig().getDataConfig() + ". Details: " + e.getMessage(), false);
150                            }
151                    }
152                    
153                    // generic checks
154                    if (endpoints.size()==0)
155                            error("No federation members specified. At least one data source is required.", true);
156                    
157                    if (queries.size()==0)
158                            error("No queries specified", true);
159                    
160                    // setup the federation
161                    try {
162                            repo = FedXFactory.initializeFederation(endpoints);
163                    } catch (FedXException e) {
164                            error("Problem occured while setting up the federation: " + e.getMessage(), false);
165                    }
166                    
167                    // initialize default prefix declarations (if the user did not specify anything)
168                    if (Config.getConfig().getPrefixDeclarations()==null) {
169                            initDefaultPrefixDeclarations();
170                    }
171                    
172                    int count=1;
173                    for (String queryString : queries) {
174                            System.out.println("Running Query " + count);
175                            try {
176                                    runQuery(queryString, count);
177                            } catch (QueryEvaluationException e) {
178                                    error("Query " + count + " could not be evaluated: \n" + e.getMessage(), false);
179                            }
180                            count++;
181                    }
182                    
183                    try {
184                            FederationManager.getInstance().shutDown();
185                    } catch (FedXException e) {
186                            System.out.println("WARN: Federation could not be shut down: " + e.getMessage());
187                    }
188                    System.out.println("Done.");
189                    System.exit(0);
190            }

195            protected void parse(String[] _args) {
196                    if (_args.length==0) {
197                            printUsage(true);
198                    }
199                    
200                    if (_args.length==1 && _args[0].equals("-help"))  {
201                            printUsage(true);               
202                    }
203                                    
204                    List<String> args = new LinkedList<String>(Arrays.asList(_args));
205                    
206                    parseConfiguaration(args, false);
207                    
208                    // initialize config
209                    try {
210                            Config.initialize(fedxConfig); // fedxConfig may be null (default values)
211                            if (prefixDeclarations!=null) {
212                                    Config.getConfig().set("prefixDeclarations", prefixDeclarations);       // override in config
213                            }
214                    } catch (FedXException e) {
215                            error("Problem occured while setting up the federation: " + e.getMessage(), false);
216                    }               
217                    
218                    parseEndpoints(args, false);
219                    parseOutput(args);
220                    parseQueries(args);
221                    
222                    if (outFolder==null) {
223                            outFolder = new SimpleDateFormat("yyyy-MM-dd HH-mm-ss").format(new Date());
224                    }
225    
226            }
227            
228            /**
229             * Parse the FedX Configuration,
230             *  1) -c path/to/fedxconfig.prop
231             *  2) -verbose [%lvl$]
232             *  3) -logtofile
233             *  4) -p path/to/prefixDeclaration.prop         
234             *  
235             *  WHERE lvl is 0=off (default), 1=INFO, 2=DEBUG, 3=TRACE
236             *  
237             * @param args
238             */
239            protected void parseConfiguaration(List<String> args, boolean printError) {
240                    String arg = args.get(0);
241                    
242                    // fedx config
243                    if (arg.equals("-c")) {
244                            readArg(args);                                                                                  // remove -c
245                            fedxConfig = readArg(args, "path/to/fedxConfig.ttl");   // remove path
246                    }
247                    
248                    // verbose
249                    else if (arg.equals("-verbose")) {
250                            verboseLevel = 1;
251                            readArg(args);
252                            try {
253                                    verboseLevel = Integer.parseInt(args.get(0));
254                                    readArg(args);
255                            } catch (Exception numberFormat) {
256                                    ; // ignore
257                            }
258                    }
259                    
260                    // logtofile
261                    else if (arg.equals("-logtofile")) {
262                            readArg(args);
263                            logtofile = true;
264                    }
265                    
266                    // prefixConfiguration
267                    else if (arg.equals("-p")) {
268                            readArg(args);
269                            prefixDeclarations = readArg(args, "path/to/prefixDeclarations.prop");
270                    }
271                    
272                    else {
273                            if (printError)
274                                    error("Unxpected Configuration Option: " + arg, false);
275                            else
276                                    return;
277                    }
278                    
279                    parseConfiguaration(args, false);
280            }

283            /**
284             * Parse the endpoints, i.e. federation members
285             *  1) SPARQL Endpoint: -s url
286             *  2) Local NativeStore: -l path/to/NativeStore
287             *  3) Dataconfig: -d path/to/dataconfig.ttl
288             *  
289             * @param args
290             */
291            protected void parseEndpoints(List<String> args, boolean printError) {
292                    String arg = args.get(0);
293                    
294                    if (arg.equals("-s")) {
295                            readArg(args);          // remove -s
296                            String url = readArg(args,"urlToSparqlEndpoint");
297                            try {
298                                    Endpoint endpoint = EndpointFactory.loadSPARQLEndpoint(url);
299                                    endpoints.add(endpoint);
300                            } catch (FedXException e) {
301                                    error("SPARQL endpoint " + url + " could not be loaded: " + e.getMessage(), false);
302                            }                       
303                    }
304                    
305                    else if (arg.equals("-l")) {
306                            readArg(args);          // remove -l
307                            String path = readArg(args,"path/to/NativeStore");
308                            try {
309                                    Endpoint endpoint = EndpointFactory.loadNativeEndpoint(path);
310                                    endpoints.add(endpoint);
311                            } catch (FedXException e) {
312                                    error("NativeStore " + path + " could not be loaded: " + e.getMessage(), false);
313                            }
314                    }
315                    
316                    else if (arg.equals("-d")) {
317                            readArg(args);          // remove -d
318                            String dataConfig = readArg(args,"path/to/dataconfig.ttl");
319                            try {
320                                    List<Endpoint> ep = EndpointFactory.loadFederationMembers(new File(dataConfig));
321                                    endpoints.addAll(ep);
322                            } catch (FedXException e) {
323                                    error("Data config '" + dataConfig + "' could not be loaded: " + e.getMessage(), false);
324                            }
325                    }
326                    
327                    else {                  
328                            if (printError)
329                                    error("Expected at least one federation member (-s, -l, -d), was: " + arg, false);
330                            else
331                                    return;                 
332                    }
333                    
334                    parseEndpoints(args, false);
335            }

337            
338            /**
339             * Parse output options
340             *  1) Format: -f {STDOUT,XML,JSON}
341             *  2) OutputFolder: -folder outputFolder 
342             * 
343             * @param args
344             */
345            protected void parseOutput(List<String> args) {
346                    
347                    String arg = args.get(0);
348                    
349                    if (arg.equals("-f")) {
350                            readArg(args);          // remove -s
351                            
352                            String format = readArg(args, "output format {STDOUT, XML, JSON}");
353                            
354                            if (format.equals("STDOUT"))
355                                    outputFormat = OutputFormat.STDOUT;
356                            else if (format.equals("JSON"))
357                                    outputFormat = OutputFormat.JSON;
358                            else if (format.equals("XML")) {
359                                    outputFormat = OutputFormat.XML;
360                            }
361                            else {
362                                    error("Unexpected output format: " + format + ". Available options: STDOUT,XML,JSON", false);
363                            }
364                    }
365                    
366                    else if (arg.equals("-folder")) {
367                            readArg(args);          // remove -folder
368                            
369                            outFolder = readArg(args, "outputFolder");
370                    }
371                    
372                    else {
373                            return;
374                    }
375                    
376                    parseOutput(args);
377            }

380            /**
381             * Parse query input
382             *  1) Querystring: -q SparqlQuery
383             *  2) File: @q path/to/QueryFile
384             *  
385             * @param args
386             */
387            protected void parseQueries(List<String> args) {
388                    String arg = args.get(0);
389                    
390                    if (arg.equals("-q")) {
391                            readArg(args);  // remove -q
392                            String query = readArg(args, "SparqlQuery");
393                            queries.add(query);
394                    }
395                    
396                    else if (arg.equals("@q")) {
397                            readArg(args);  // remove @q
398                            String queryFile = readArg(args, "path/to/queryFile");
399                            try {   
400                                    List<String> q = QueryStringUtil.loadQueries(queryFile);
401                                    queries.addAll(q);
402                            } catch (IOException e) {
403                                    error("Error loading query file '" + queryFile + "': " + e.getMessage(), false);
404                            }
405                    }
406                    
407                    else {
408                            error("Unexpected query argument: " + arg, false);
409                    }
410                    
411                    if (args.size()>0)
412                            parseQueries(args);
413            }
414            
415            protected String readArg(List<String> args, String... expected) {
416                    if (args.size()==0)
417                            error("Unexpected end of args, expected: " + expected, false);
418                    return args.remove(0);
419            }

421            /**
422             * initializes default prefix declarations from com.fluidops.fedx.commonPrefixesCli.prop
423             */
424            protected void initDefaultPrefixDeclarations() {
425                    
426                    QueryManager qm = FederationManager.getInstance().getQueryManager();
427                    Properties props = new Properties();
428                    try     {
429                            props.load(CLI.class.getResourceAsStream("/com/fluidops/fedx/commonPrefixesCli.prop"));
430                    } catch (IOException e) {
431                            throw new FedXRuntimeException("Error loading prefix properties: " + e.getMessage());
432                    }
433                    
434                    for (String ns : props.stringPropertyNames()) {
435                            qm.addPrefixDeclaration(ns, props.getProperty(ns));     // register namespace/prefix pair
436                    }
437            }

439            protected void runQuery(String queryString, int queryId) throws QueryEvaluationException {
440                            
441                    TupleQuery query;
442                    try {
443                            query = QueryManager.prepareTupleQuery(queryString);
444                    } catch (MalformedQueryException e) {
445                            throw new QueryEvaluationException("Query is malformed: " + e.getMessage());
446                    } 
447                    int count = 0;
448                    long start = System.currentTimeMillis();
449                    
450                    TupleQueryResult res = query.evaluate();
451                                    
452                    if (outputFormat == OutputFormat.STDOUT) {
453                            while (res.hasNext()) {
454                                    System.out.println(res.next());
455                                    count++;
456                            }
457                    }
458                    
459                    else if (outputFormat == OutputFormat.JSON) {
460                            
461                            File out = new File("results/"+ outFolder + "/q_" + queryId + ".json");
462                            out.getParentFile().mkdirs();
463                            
464                            System.out.println("Results are being written to " + out.getPath());
465                            
466                            try {
467                                    SPARQLResultsJSONWriter w = new SPARQLResultsJSONWriter(new FileOutputStream(out));
468                                    w.startQueryResult(res.getBindingNames());
469                                    
470                                    while (res.hasNext()) {
471                                            w.handleSolution(res.next());
472                                            count++;
473                                    }
474                                    
475                                    w.endQueryResult();
476                            } catch (IOException e) {
477                                    error("IO Error while writing results of query " + queryId + " to JSON file: " + e.getMessage(), false);
478                            } catch (TupleQueryResultHandlerException e) {
479                                    error("Tuple result error while writing results of query " + queryId + " to JSON file: " + e.getMessage(), false);
480                            }
481                    }
482                    
483                    else if (outputFormat == OutputFormat.XML) {
484                            
485                            File out = new File("results/" + outFolder + "/q_" + queryId + ".xml");
486                            out.getParentFile().mkdirs();
487                            
488                            System.out.println("Results are being written to " + out.getPath());
489                            
490                            try {
491                                    SPARQLResultsXMLWriter w = new SPARQLResultsXMLWriter(new FileOutputStream(out));
492                                    w.startQueryResult(res.getBindingNames());
493                                    
494                                    while (res.hasNext()) {
495                                            w.handleSolution(res.next());
496                                            count++;
497                                    }
498                                    
499                                    w.endQueryResult();
500                                    
501                            } catch (IOException e) {
502                                    error("IO Error while writing results of query " + queryId + " to XML file: " + e.getMessage(), false);
503                            } catch (TupleQueryResultHandlerException e) {
504                                    error("Tuple result error while writing results of query " + queryId + " to JSON file: " + e.getMessage(), false);
505                            }
506                    }
507                    
508                    long duration = System.currentTimeMillis() - start;             // the duration in ms
509                    
510                    System.out.println("Done query " + queryId + ": duration=" + duration + "ms, results=" + count);
511            }

575            /**
576             * Activate logging if -verbose is enabled.
577             * 
578             * Verbose level: 0=off (default), 1=INFO, 2=DEBUG, 3=ALL
579             */
580            protected void configureLogging() {
581                    
582                    //Logger rootLogger = Logger.getRootLogger();
583                    Logger l = Logger.getLogger("com.fluidops.fedx");
584                    Logger rootLogger = Logger.getRootLogger();
585                    if (verboseLevel>0) {
586                            //Logger pkgLogger = rootLogger.getLoggerRepository().getLogger("com.fluidops.fedx"); 
587                            if (verboseLevel==1) {
588                                    rootLogger.setLevel(Level.INFO);
589                                    l.setLevel(Level.INFO);
590                            } else if (verboseLevel==1) {
591                                    rootLogger.setLevel(Level.DEBUG);
592                                    l.setLevel(Level.DEBUG);
593                            }       else if (verboseLevel>2) {
594                                    rootLogger.setLevel(Level.ALL);
595                                    l.setLevel(Level.ALL);
596                            }
597                                    
598                            if (logtofile) {
599                                    try {
600                                            l.addAppender( new FileAppender(new PatternLayout("%5p %d{yyyy-MM-dd hh:mm:ss} [%t] (%F:%L) - %m%n"), "logs/fedx_cli.log"));
601                                    } catch (IOException e) {
602                                            System.out.println("WARN: File Logging could not be initialized: " + e.getMessage());
603                                    }                               
604                            } else {
605                                    l.addAppender(new ConsoleAppender(new PatternLayout("%5p [%t] (%F:%L) - %m%n")));
606                            }
607                    }               
608            }
609            
610            protected void configureRootLogger() {
611                    
612                    Logger rootLogger = Logger.getRootLogger();
613                    if (!rootLogger.getAllAppenders().hasMoreElements()) {
614                            rootLogger.setLevel(Level.ALL); 
615                            rootLogger.addAppender(new NullAppender() );      
616                    }
617            }
