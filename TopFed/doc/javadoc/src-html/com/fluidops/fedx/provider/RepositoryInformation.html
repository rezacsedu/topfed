018    package com.fluidops.fedx.provider;
019    
020    import java.util.Properties;
021    
022    import com.fluidops.fedx.structures.Endpoint.EndpointType;
023    
024    public class RepositoryInformation {
025    
026            protected Properties props = new Properties();
027            private EndpointType type = EndpointType.Other; // the endpoint type, default Other
028            
029            public RepositoryInformation(String id, String name, String location, EndpointType type) {
030                    props.setProperty("id", id);
031                    props.setProperty("name", name);
032                    props.setProperty("location", location);
033                    this.type = type;
034            }
035            
036            protected RepositoryInformation(EndpointType type) {
037                    this.type = type;
038            }
039            
040            public String getId() {
041                    return props.getProperty("id");
042            }
043            
044            public String getName() {
045                    return props.getProperty("name");
046            }
047            
048            public String getLocation() {
049                    return props.getProperty("location");
050            }
051            
052            public EndpointType getType() {
053                    return type;
054            }
055            
056            public String get(String key) {
057                    return props.getProperty(key);
058            }
059            
060            public String get(String key, String def) {
061                    return props.getProperty(key, def);
062            }
063            
064            public void setProperty(String key, String value) {
065                    props.setProperty(key, value);
066            }
067    }
